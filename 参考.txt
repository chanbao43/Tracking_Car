void delay_ms(unsigned int n)
{
    unsigned int i,j;
    for(i=n; i>0; i--)
        for(j=114; j>0; j--);
}

// ??????? - ?????

{

// PWM????
void Left_Motor_PWM_Adjust(void)
{
   if(Left_moto_stop)
   {
       if(Left_PWM_Value<=Left_Drive_Value)
           Left_Motor_PWM=1;
       else
           Left_Motor_PWM=0;
       if(Left_PWM_Value>=20)
           Left_PWM_Value=0;
   }
   else
   {
       Left_Motor_PWM=0;
   }
}
void Right_Motor_PWM_Adjust(void)
{
  if(Right_moto_stop)
   {
        if(Right_PWM_Value<=Right_Drive_Value)
            Right_Motor_PWM=1;
        else
            Right_Motor_PWM=0;
        if(Right_PWM_Value>=20)
            Right_PWM_Value=0;
   }
   else
   {
        Right_Motor_PWM=0;
   }
}

void Check_Special_Turn(void)
{
    // ???????????
    if(Turn_Flag == 1)
    {
        // ????10?11?14?15?????90°
        if(Line == 10 || Line == 11 || Line == 14 || Line == 15)
        {
            if(Special_Turn_Flag == 0)  // ??????
            {
                Special_Turn_Flag = 1;
                Stop();
                delay_ms(50);
                Turn_Left_90();
                delay_ms(50);
            }
        }
        else
        {
            Special_Turn_Flag = 0;  // ????
        }
    }
}

// ??????
void Track(void)
{
    if(Turning) return;

    // ??????????
    Check_Special_Turn();
    
    if(Left_IRSenor_Track==0 && Right_IRSenor_Track==0)
    {
        GoForward();
        Black_Line_Detected = 1;
        Debounce_Cnt = 0;
    }
    else if(Left_IRSenor_Track==0 && Right_IRSenor_Track==1)
    {
        GoRight();
    }
    else if(Left_IRSenor_Track==1 && Right_IRSenor_Track==0)
    {
        GoLeft();
    }
    else if(Left_IRSenor_Track==1 && Right_IRSenor_Track==1)
    {
        GoForward();
        if(Black_Line_Detected == 1)
        {
            Debounce_Cnt++;
            if(Debounce_Cnt > 10)
            {
                Line++;
                Update_Display();  // ????
                Black_Line_Detected = 0;
                Debounce_Cnt = 0;
                
                // ???:???8???
                if(Turn_Flag == 0 && Line == 8)
                {
                    Stop();
                    delay_ms(100);
                    Turn_180();
                    Turn_Flag = 1;  // ??????
                    delay_ms(100);
                }
                // ???:???16???
                else if(Turn_Flag == 1 && Line == 16)
                {
                    Stop();
                    while(1)
                    {
                        LEDTube_Show();  // ??????16
                    }
                }
            }
        }
    }
}
